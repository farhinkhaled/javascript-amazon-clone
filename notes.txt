- mocks -> replace a method with a fake version 
- spyOn() records everytime a method is used 
-       // localStorage.setItem will be replaced with a fake version and addToCart won't save product in the actual cart 
        spyOn(localStorage, 'setItem');

        // mock locaStorage.getItem to get an empty cart
        spyOn(localStorage, 'getItem').and.callFake(() => {
            return JSON.stringify([]);
        });

        // an empty cart will be found when the cart will be reloaded
        loadFromStorage();

        // now when a product will be added to an empty cart the cart length will be equal to one 
        addToCart('e43638ce-6aa0-4b85-b27f-e1d07eb678c6');
        expect(cart.length).toEqual(1);

        // expect localStorage.setItem has been called once 
        // this only works when localStorage.setItem have been mocked with spyOn
        expect(localStorage.setItem).toHaveBeenCalledTimes(1);
